// Shared client helpers (esc, toggleInfo), included by both pages via include(). // IMPORTANT: this file must contain NO literal ampersand, less-than, or greater-than // character anywhere (not in code, strings, comments, or a loop condition). The // include pipeline HTML-escapes a literal less-than into its entity, which silently // breaks whatever it touches (a regex char-set, or a comparison operator). So esc // builds its special characters with String.fromCharCode and avoids comparison // operators by mapping over the characters. function esc(s) { var AMP = String.fromCharCode(38); // ampersand var map = {}; map[AMP] = AMP + 'amp;'; map[String.fromCharCode(60)] = AMP + 'lt;'; // less-than map[String.fromCharCode(62)] = AMP + 'gt;'; // greater-than map[String.fromCharCode(34)] = AMP + 'quot;'; // double quote map[String.fromCharCode(39)] = AMP + '#39;'; // single quote return String(s).split('').map(function(ch) { return map.hasOwnProperty(ch) ? map[ch] : ch; }).join(''); } // Collapsible Rules/Info panel (static, independent of the polling loop). function toggleInfo(btn) { const open = btn.getAttribute('aria-expanded') === 'true'; btn.setAttribute('aria-expanded', String(!open)); const body = btn.parentNode.querySelector('.info-body'); if (body) body.hidden = open; }
